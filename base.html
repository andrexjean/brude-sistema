""" BRUDE - Parser Nubank Suporta PDF digital e CSV exportado pelo app. Estratégia: tenta CSV primeiro (mais confiável), fallback para PDF. """ import re import csv import hashlib import uuid from pathlib import Path from datetime import datetime from typing import Optional import pdfplumber # ── Padrões Nubank ──────────────────────────────────────────── # Nubank PDF digital: linha típica "07 ABR SHOPEE*PRODUTO 123,45" _NU_LINHA = re.compile( r'^(\d{2})\s+(JAN|FEV|MAR|ABR|MAI|JUN|JUL|AGO|SET|OUT|NOV|DEZ)\s+' r'(.+?)\s+([\d\.]+,\d{2})$', re.IGNORECASE ) _MESES_PT = { 'JAN': '01', 'FEV': '02', 'MAR': '03', 'ABR': '04', 'MAI': '05', 'JUN': '06', 'JUL': '07', 'AGO': '08', 'SET': '09', 'OUT': '10', 'NOV': '11', 'DEZ': '12' } # Padrão de parcelamento Nubank: "SHOPEE 3/10" ou "AMAZON PARC 5/12" _PARCELA_NU = re.compile(r'(\d{1,2})/(\d{1,2})') def _parse_valor_br(valor_str: str) -> float: """'1.234,56' → 1234.56""" return float(valor_str.replace('.', '').replace(',', '.')) def _gerar_hash(data: str, descricao: str, valor: float, membro: str) -> str: """SHA256 para deduplicação. Mesma transação = mesmo hash.""" chave = f"{data}|{descricao.upper().strip()}|{valor:.2f}|{membro}" return hashlib.sha256(chave.encode()).hexdigest()[:32] def _detectar_parcela(descricao: str) -> tuple[Optional[int], Optional[int]]: match = _PARCELA_NU.search(descricao) if match: atual = int(match.group(1)) total = int(match.group(2)) if 1 <= atual <= total <= 72 and total >= 2: return atual, total return None, None def _inferir_ano(mes: str, mes_ref: Optional[str]) -> int: """ Infere o ano da transação. Nubank agrupa em fatura: transações de meses anteriores aparecem numa fatura futura (ex: compra em mar/26 aparece na fatura abr/26). """ if mes_ref: ano_ref = int(mes_ref[:4]) mes_ref_num = int(mes_ref[5:7]) mes_num = int(mes) # Se o mês da transação é posterior ao mês de referência, # pertence ao ano anterior if mes_num > mes_ref_num: return ano_ref - 1 return ano_ref return datetime.now().year # ── Parser CSV (preferencial) ───────────────────────────────── def parse_nubank_csv( filepath: str, membro: str, mes_referencia: Optional[str] = None ) -> list[dict]: """ Parseia CSV exportado pelo app Nubank. Formato esperado: date,category,title,amount ou: Data,Descrição,Valor """ transacoes = [] filepath = Path(filepath) with open(filepath, encoding='utf-8-sig') as f: reader = csv.DictReader(f) headers = [h.lower().strip() for h in (reader.fieldnames or [])] for row in reader: row_lower = {k.lower().strip(): v for k, v in row.items()} try: # Suporta múltiplos formatos de CSV do Nubank if 'date' in row_lower: data_raw = row_lower.get('date', '') descricao = row_lower.get('title', '') or row_lower.get('description', '') valor_raw = row_lower.get('amount', '0') elif 'data' in row_lower: data_raw = row_lower.get('data', '') descricao = row_lower.get('descrição', '') or row_lower.get('descricao', '') valor_raw = row_lower.get('valor', '0') else: continue # Normaliza data → ISO data = _normalizar_data(data_raw) if not data: continue # Valor valor_str = str(valor_raw).replace(' ', '') if ',' in valor_str and '.' in valor_str: valor = _parse_valor_br(valor_str) elif ',' in valor_str: valor = float(valor_str.replace(',', '.')) else: valor = float(valor_str) # Ignorar valores negativos (estornos/pagamentos de fatura) if valor <= 0: continue parcela_atual, parcela_total = _detectar_parcela(descricao) grupo_parcela = str(uuid.uuid4()) if parcela_total else None mes_ref = mes_referencia or _inferir_mes_ref(data) hash_id = _gerar_hash(data, descricao, valor, membro) transacoes.append({ 'hash_dedup': hash_id, 'fonte': 'nubank', 'membro_id': None, # resolvido no pipeline 'membro_nome': membro, 'data': data, 'descricao_original': descricao.strip(), 'descricao_norm': None, # preenchido pelo normalizer 'valor': round(valor, 2), 'categoria_id': None, # preenchido pelo classifier 'categoria_sugerida': None, 'confianca_norm': 0.0, 'classificacao_auto': 1, 'confianca': 0.0, 'parcela_atual': parcela_atual, 'parcela_total': parcela_total, 'grupo_parcela': grupo_parcela, 'mes_referencia': mes_ref, 'cartao': 'Nubank', }) except (ValueError, KeyError) as e: print(f" ⚠️ Linha ignorada: {row} — {e}") continue print(f"📄 Nubank CSV: {len(transacoes)} transações lidas de {filepath.name}") return transacoes # ── Parser PDF ───────────────────────────────────────────────── def parse_nubank_pdf( filepath: str, membro: str, mes_referencia: Optional[str] = None ) -> list[dict]: """ Parseia PDF digital do Nubank. Abordagem: extrai texto por página, identifica linhas de transação. Robusto a quebras de linha e variações de layout. """ transacoes = [] filepath = Path(filepath) try: with pdfplumber.open(filepath) as pdf: texto_completo = [] for page in pdf.pages: texto = page.extract_text(x_tolerance=3, y_tolerance=3) if texto: texto_completo.append(texto) # Detecta mês de referência no PDF se não foi informado if not mes_referencia: for texto in texto_completo: from core.normalizer import extrair_mes_referencia mes_referencia = extrair_mes_referencia(texto) if mes_referencia: break # Processa linha a linha linhas_brutas = '\n'.join(texto_completo).splitlines() linhas_limpas = _preprocessar_linhas(linhas_brutas) for linha in linhas_limpas: trans = _parse_linha_nubank(linha, membro, mes_referencia) if trans: transacoes.append(trans) except Exception as e: print(f"❌ Erro ao ler PDF {filepath.name}: {e}") return [] print(f"📄 Nubank PDF: {len(transacoes)} transações lidas de {filepath.name}") return transacoes def _preprocessar_linhas(linhas: list[str]) -> list[str]: """ Une linhas quebradas e remove linhas irrelevantes. O Nubank às vezes quebra descrições longas em 2 linhas. """ resultado = [] buffer = "" for linha in linhas: linha = linha.strip() if not linha: if buffer: resultado.append(buffer) buffer = "" continue # Linha começa com data (dd MES) → nova transação if re.match(r'^\d{2}\s+(JAN|FEV|MAR|ABR|MAI|JUN|JUL|AGO|SET|OUT|NOV|DEZ)', linha, re.IGNORECASE): if buffer: resultado.append(buffer) buffer = linha elif buffer and not re.search(r'[\d\.]+,\d{2}$', buffer): # Linha anterior não tem valor ainda, pode ser continuação buffer += ' ' + linha else: if buffer: resultado.append(buffer) buffer = linha if buffer: resultado.append(buffer) return resultado def _parse_linha_nubank( linha: str, membro: str, mes_referencia: Optional[str] ) -> Optional[dict]: """ Parseia uma linha de transação do PDF Nubank. Formato: "07 ABR SHOPEE*PRODUTO 3/10 R$ 123,45" """ match = _NU_LINHA.match(linha.strip()) if not match: return None dia, mes_str, descricao, valor_str = match.groups() # Ignora linhas de header/total if any(kw in descricao.upper() for kw in ['TOTAL', 'PAGAMENTO', 'SALDO', 'ANTERIOR', 'LIMITE']): return None mes_num = _MESES_PT.get(mes_str.upper(), '01') ano = _inferir_ano(mes_num, mes_referencia) data = f"{ano}-{mes_num}-{dia.zfill(2)}" valor = _parse_valor_br(valor_str) if valor <= 0: return None parcela_atual, parcela_total = _detectar_parcela(descricao) grupo_parcela = str(uuid.uuid4()) if parcela_total else None mes_ref = mes_referencia or _inferir_mes_ref(data) hash_id = _gerar_hash(data, descricao, valor, membro) return { 'hash_dedup': hash_id, 'fonte': 'nubank', 'membro_id': None, 'membro_nome': membro, 'data': data, 'descricao_original': descricao.strip(), 'descricao_norm': None, 'valor': round(valor, 2), 'categoria_id': None, 'categoria_sugerida': None, 'confianca_norm': 0.0, 'classificacao_auto': 1, 'confianca': 0.0, 'parcela_atual': parcela_atual, 'parcela_total': parcela_total, 'grupo_parcela': grupo_parcela, 'mes_referencia': mes_ref, 'cartao': 'Nubank', } def _normalizar_data(data_str: str) -> Optional[str]: """Normaliza vários formatos de data para ISO 8601.""" formatos = ['%Y-%m-%d', '%d/%m/%Y', '%d-%m-%Y', '%m/%d/%Y'] for fmt in formatos: try: return datetime.strptime(data_str.strip(), fmt).strftime('%Y-%m-%d') except ValueError: continue return None def _inferir_mes_ref(data_iso: str) -> str: """Infere mês de referência a partir da data da transação.""" try: dt = datetime.fromisoformat(data_iso) return f"{dt.year}-{dt.month:02d}" except Exception: return datetime.now().strftime('%Y-%m') def parse_nubank( filepath: str, membro: str, mes_referencia: Optional[str] = None ) -> list[dict]: """ Entry point: detecta automaticamente se é PDF ou CSV. """ ext = Path(filepath).suffix.lower() if ext == '.csv': return parse_nubank_csv(filepath, membro, mes_referencia) elif ext == '.pdf': return parse_nubank_pdf(filepath, membro, mes_referencia) else: raise ValueError(f"Formato não suportado: {ext}. Use PDF ou CSV.")
